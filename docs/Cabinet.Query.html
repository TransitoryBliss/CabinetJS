Example
var query = MyModel.query().where({
  username: "John", 
  email: {
    equals: "john@mail.com"
  },
  age: {
    greaterThan: 18,
    lessThan: 24
  }
});
// results in the following query object
// {
//   username: { 
//	   equals: "John"
//   },
//   email: {
//     equals: "john@mail.com"
//   }, 
//   age: {
//     greaterThan: 18,
